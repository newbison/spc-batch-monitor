```python
from abc import ABC, abstractmethod
import pandas as pd


class DataRepository(ABC):
    @abstractmethod
    def load_all(self) -> pd.DataFrame:
        """Return all measurement data as a DataFrame."""

    @abstractmethod
    def get_for_parameter(self, parameter: str) -> pd.DataFrame:
        """Return rows for a single parameter (pH, IV, viscosity)."""

    @abstractmethod
    def append_batch(self, batch_id: str, date: str, formula: str,
                     measurements: dict[str, dict]) -> None:

            'IV':  {'reps': [0.952, 0.947, 0.961], 'lower_spec': 0.90, 'upper_spec': 1.10},
```

```python
import pandas as pd
from pathlib import Path

from .base import DataRepository
from .file_reader import read_csv, write_csv


class CsvRepository(DataRepository):
    COLUMNS = ["date", "batch_id", "formula", "parameter", "rep1", "rep2", "rep3",
               "lower_spec", "upper_spec"]

    def __init__(self, filepath: Path):
        self.filepath = filepath

    def load_all(self) -> pd.DataFrame:
        if not self.filepath.exists():
            return pd.DataFrame(columns=self.COLUMNS)
        return read_csv(self.filepath)

    def get_for_parameter(self, parameter: str) -> pd.DataFrame:
        df = self.load_all()
        if df.empty:
            return df
        return df[df["parameter"] == parameter].copy().reset_index(drop=True)

    def get_formulas(self) -> list[str]:
        df = self.load_all()
        if df.empty:
            return []
        return df["formula"].unique().tolist()

    def get_specs_for_formula(self, formula: str, parameter: str) -> dict | None:
        df = self.load_all()
        if df.empty:
            return None
        match = df[(df["formula"] == formula) & (df["parameter"] == parameter)]
        if match.empty:
            return None
        row = match.iloc[-1]
        return {"lower_spec": row["lower_spec"], "upper_spec": row["upper_spec"]}

    def append_batch(self, batch_id: str, date: str, formula: str,
                     measurements: dict[str, dict]) -> None:
        df = self.load_all()
        rows = []
        for param, entry in measurements.items():
            rows.append({
                "date": date,
                "batch_id": batch_id,
                "formula": formula,
                "parameter": param,
                "rep1": entry["reps"][0],
                "rep2": entry["reps"][1],
                "rep3": entry["reps"][2],
                "lower_spec": entry["lower_spec"],
                "upper_spec": entry["upper_spec"],
            })
        df = pd.concat([df, pd.DataFrame(rows)], ignore_index=True)
        write_csv(df, self.filepath)
```

```python
"""Generate dummy adhesive batch data for SPC development."""
import numpy as np
import pandas as pd
from pathlib import Path

np.random.seed(42)

# Each formula has its own specs
FORMULAS = {
    "Adhesive A": {
        "pH":        {"target": 7.00, "sigma_between": 0.30, "sigma_within": 0.08, "lsl": 5.50, "usl": 8.50},
        "IV":        {"target": 1.00, "sigma_between": 0.04, "sigma_within": 0.01, "lsl": 0.90, "usl": 1.10},
        "viscosity": {"target": 10500, "sigma_between": 1500, "sigma_within": 300, "lsl": 6000, "usl": 15000},
    },
    "Adhesive B": {
        "pH":        {"target": 6.50, "sigma_between": 0.20, "sigma_within": 0.05, "lsl": 5.00, "usl": 8.00},
        "IV":        {"target": 0.95, "sigma_between": 0.03, "sigma_within": 0.01, "lsl": 0.85, "usl": 1.05},
        "viscosity": {"target": 9500,  "sigma_between": 1200, "sigma_within": 250, "lsl": 5000, "usl": 14000},
    },
}

N_BATCHES = 30
OUTPUT = Path(__file__).parent / "adhesive_batches.csv"


def generate_batch(batch_id: str, date: str, formula: str) -> list[dict]:
    rows = []
    cfg = FORMULAS[formula]
    for param, p in cfg.items():
        batch_mean = np.random.normal(p["target"], p["sigma_between"])
        reps = [
            round(np.random.normal(batch_mean, p["sigma_within"]),
                  3 if param != "viscosity" else 0)
            for _ in range(3)
        ]
        rows.append({
            "date": date,
            "batch_id": batch_id,
            "formula": formula,
            "parameter": param,
            "rep1": reps[0],
            "rep2": reps[1],
            "rep3": reps[2],
            "lower_spec": p["lsl"],
            "upper_spec": p["usl"],
        })
    return rows


def main():
    start = pd.Timestamp("2025-01-02")
    formulas = list(FORMULAS.keys())
    all_rows = []
    for i in range(N_BATCHES):
        batch_id = f"ADH-{i+1:03d}"
        date = (start + pd.Timedelta(days=i)).strftime("%Y-%m-%d")
        # Alternate formulas every 15 batches
        formula = formulas[i // 15]
        all_rows.extend(generate_batch(batch_id, date, formula))

    df = pd.DataFrame(all_rows)
    df.to_csv(OUTPUT, index=False)
    print(f"Wrote {len(df)} rows ({N_BATCHES} batches) to {OUTPUT}")
    print(f"Formulas: {df['formula'].unique().tolist()}")
    print(df.head(9))


if __name__ == "__main__":
    main()
```

```python
import numpy as np
import pandas as pd

# Constants for subgroup size n=3
A2 = 1.023
D3 = 0.0
D4 = 2.574


def compute_xbar_r(df: pd.DataFrame) -> dict:
    reps = df[["rep1", "rep2", "rep3"]].values.astype(float)
    xbar = np.mean(reps, axis=1)
    r = np.max(reps, axis=1) - np.min(reps, axis=1)

    Xbarbar = float(np.mean(xbar))
    Rbar = float(np.mean(r))

    return {
        "xbar": xbar,
        "r": r,
        "Xbarbar": Xbarbar,
        "Rbar": Rbar,
        "UCLx": Xbarbar + A2 * Rbar,
        "LCLx": Xbarbar - A2 * Rbar,
        "CLx": Xbarbar,
        "UCLr": D4 * Rbar,
        "LCLr": D3 * Rbar,
        "CLr": Rbar,
    }
```

```python
import numpy as np


def check_rules(xbar: np.ndarray, ucl: float, lcl: float, cl: float) -> list[dict]:
    n = len(xbar)
    sigma = (ucl - cl) / 3.0
    violations = []

    for i in range(n):
        # Rule 1: single point beyond 3-sigma
        if xbar[i] > ucl or xbar[i] < lcl:
            violations.append({
                "batch_index": i, "rule": 1,
                "description": f"Point beyond 3σ (x̄={xbar[i]:.3f})",
            })

    for i in range(n - 2):
        # Rule 2: 2 of 3 consecutive beyond 2-sigma (same side)
        window = xbar[i:i + 3]
        above_2sigma = np.sum(window > cl + 2 * sigma) >= 2
        below_2sigma = np.sum(window < cl - 2 * sigma) >= 2
        if above_2sigma or below_2sigma:
            violations.append({
                "batch_index": i + 2, "rule": 2,
                "description": "2 of 3 points beyond 2σ",
            })

    for i in range(n - 7):
        # Rule 4: 8 consecutive on same side of centerline
        window = xbar[i:i + 8]
        if np.all(window > cl) or np.all(window < cl):
            violations.append({
                "batch_index": i + 7, "rule": 4,
                "description": "8 consecutive points on same side of centerline",
            })

    for i in range(n - 5):
        # Trending: 6 consecutive increasing or decreasing
        window = xbar[i:i + 6]
        diffs = np.diff(window)
        if np.all(diffs > 0):
            violations.append({
                "batch_index": i + 5, "rule": 5,
                "description": "6 consecutive points trending up",
            })
        if np.all(diffs < 0):
            violations.append({
                "batch_index": i + 5, "rule": 5,
                "description": "6 consecutive points trending down",
            })

    return violations
```

```python
import numpy as np
import math
from scipy import stats


def compute_capability(xbar: np.ndarray, lsl: float, usl: float) -> dict:
    mean = float(np.mean(xbar))
    sigma_overall = float(np.std(xbar, ddof=1))

    has_lsl = not math.isnan(lsl)
    has_usl = not math.isnan(usl)

    if sigma_overall == 0:
        return {
            "sigma_overall": 0.0,
            "Pp": float("inf") if (has_lsl and has_usl) else float("nan"),
            "Ppk": float("inf"),
            "mean": mean, "ppm_usl": 0.0, "ppm_lsl": 0.0, "total_ppm": 0.0,
        }

    # Pp: needs both limits
    if has_lsl and has_usl:
        Pp = (usl - lsl) / (6 * sigma_overall)
    else:
        Pp = float("nan")

    # Ppk: uses whichever limit(s) are available
    pk_candidates = []
    if has_usl:
        pk_candidates.append((usl - mean) / (3 * sigma_overall))
    if has_lsl:
        pk_candidates.append((mean - lsl) / (3 * sigma_overall))
    Ppk = min(pk_candidates) if pk_candidates else float("nan")

    # PPM
    ppm_usl = (1 - stats.norm.cdf(usl, loc=mean, scale=sigma_overall)) * 1_000_000 if has_usl else 0.0
    ppm_lsl = stats.norm.cdf(lsl, loc=mean, scale=sigma_overall) * 1_000_000 if has_lsl else 0.0

    return {
        "sigma_overall": sigma_overall,
        "Pp": Pp,
        "Ppk": Ppk,
        "mean": mean,
        "ppm_usl": ppm_usl,
        "ppm_lsl": ppm_lsl,
        "total_ppm": ppm_usl + ppm_lsl,
    }
```

```python
from plotly.subplots import make_subplots
import numpy as np


def build_xbar_r_chart(
    limits: dict,
    violations: list[dict],
    parameter_name: str,

            name="X̄", marker=dict(size=6, color="#2196F3"),

        row=1, col=1,
    )
    for line_val, color, label in [
        (limits["UCLx"], "#f44336", f'UCL ({limits["UCLx"]:.3f})'),
        (limits["LCLx"], "#f44336", f'LCL ({limits["LCLx"]:.3f})'),
        (limits["CLx"], "#4CAF50", f'CL ({limits["CLx"]:.3f})'),
    ]:
        fig.add_hline(y=line_val, line_dash="dash", line_color=color,
                      annotation_text=label, row=1, col=1)

    # Mark violations
    violation_indices = {v["batch_index"] for v in violations if v["rule"] == 1}
    viol_x = [x_labels[i] for i in violation_indices if i < n]
    viol_y = [limits["xbar"][i] for i in violation_indices if i < n]
    if viol_x:
        fig.add_trace(
            go.Scatter(x=viol_x, y=viol_y, mode="markers",
                       marker=dict(color="#f44336", size=10, symbol="x"),
                       name="Rule 1 violation"),
            row=1, col=1,
        )

    # --- R trace ---
    fig.add_trace(
        go.Scatter(
            x=x_labels, y=limits["r"], mode="lines+markers",
            name="R", marker=dict(size=6, color="#4CAF50"),
        ),
        row=2, col=1,
    )
    for line_val, color, label in [
        (limits["UCLr"], "#f44336", f'UCL ({limits["UCLr"]:.3f})'),
        (limits["CLr"], "#4CAF50", f'R̄ ({limits["CLr"]:.3f})'),
    ]:
        fig.add_hline(y=line_val, line_dash="dash", line_color=color,
                      annotation_text=label, row=2, col=1)

    fig.update_layout(
        title=f"X-bar & R Chart — {parameter_name}",
        height=650, showlegend=True,
    )
    fig.update_yaxes(title_text="X̄", row=1, col=1)
    fig.update_yaxes(title_text="R", row=2, col=1)
    fig.update_xaxes(title_text="Batch", row=2, col=1)

    return fig
```

```python
import plotly.graph_objects as go

def build_capability_chart(
    xbar: np.ndarray,

    target: float | None,

        title=f"Process Capability — {parameter_name}",
        xaxis_title="X̄ value",
        yaxis_title="Frequency",
```

```python
import streamlit as st
import math
from datetime import date

from data_access.repository import CsvRepository


def render_data_entry(repo: CsvRepository, parameter: str):
    st.header("🧪 Batch Data Entry")
    st.caption("Enter measurements for each parameter. 3 replicate tests per batch.")

    # --- Formula selection ---
    existing_formulas = repo.get_formulas()
    formula_options = existing_formulas + ["+ New Formula"]
    selected = st.selectbox("Formula", formula_options, key="entry_formula")

    if selected == "+ New Formula":
        formula = st.text_input("New formula name", key="entry_new_formula")
        if formula:
            existing_formulas.append(formula)
    else:
        formula = selected

    # Auto-populate specs from prior rows of same formula, or manual entry
    prior_specs = repo.get_specs_for_formula(formula, parameter) if formula else None

    col1, col2 = st.columns(2)
    with col1:
        batch_id = st.text_input("Batch ID", value="ADH-", key="entry_batch_id")
    with col2:
        batch_date = st.date_input("Date", value=date.today(), key="entry_date")

    st.divider()

    params = ["pH", "IV", "viscosity"]
    measurements = {}
    cols = st.columns(len(params))

    for i, param in enumerate(params):
        ps = prior_specs if prior_specs else {"lower_spec": 0.0, "upper_spec": 0.0}

        with cols[i]:
            st.markdown(f"**{param}**")

            # LSL with "no limit" toggle
            sc1, sc2 = st.columns([3, 1])
            with sc1:
                low_default = float(ps["lower_spec"]) if not math.isnan(float(ps["lower_spec"])) else 0.0
                low = st.number_input(f"LSL", value=low_default,
                                      key=f"{param}_lsl", format="%.4f" if param == "IV" else "%.2f")
            with sc2:
                no_lsl = st.checkbox("None", key=f"{param}_no_lsl",
                                     value=math.isnan(float(ps["lower_spec"])))

            # USL with "no limit" toggle
            sc3, sc4 = st.columns([3, 1])
            with sc3:
                high_default = float(ps["upper_spec"]) if not math.isnan(float(ps["upper_spec"])) else 0.0
                high = st.number_input(f"USL", value=high_default,
                                       key=f"{param}_usl", format="%.4f" if param == "IV" else "%.2f")
            with sc4:
                no_usl = st.checkbox("None", key=f"{param}_no_usl",
                                     value=math.isnan(float(ps["upper_spec"])))

            lower_spec = float("nan") if no_lsl else low
            upper_spec = float("nan") if no_usl else high

            target = (low + high) / 2 if not (no_lsl or no_usl) else (low if not no_lsl else high)
            rep1 = st.number_input(f"{param} Rep 1", key=f"{param}_r1", value=target,
                                   format="%.4f" if param == "IV" else "%.2f")
            rep2 = st.number_input(f"{param} Rep 2", key=f"{param}_r2", value=target,
                                   format="%.4f" if param == "IV" else "%.2f")
            rep3 = st.number_input(f"{param} Rep 3", key=f"{param}_r3", value=target,
                                   format="%.4f" if param == "IV" else "%.2f")

            avg = (rep1 + rep2 + rep3) / 3
            rng = max(rep1, rep2, rep3) - min(rep1, rep2, rep3)

            # Check against whichever spec limits exist
            in_spec = True
            if not no_lsl and avg < lower_spec:
                in_spec = False
            if not no_usl and avg > upper_spec:
                in_spec = False
            status = "✅" if in_spec else "❌"

            st.metric(f"X̄", f"{avg:.3f} {status}")
            st.metric(f"Range", f"{rng:.3f}")

            measurements[param] = {
                "reps": [rep1, rep2, rep3],
                "lower_spec": lower_spec,
                "upper_spec": upper_spec,
            }

    if st.button("💾 Save Batch", type="primary"):
        if not formula:
            st.error("Please select or enter a formula name.")
        else:
            repo.append_batch(batch_id, batch_date.isoformat(), formula, measurements)
            st.success(f"Batch {batch_id} saved for {batch_date.isoformat()}")
            st.rerun()

    st.divider()
    st.subheader("Recent Batches")
    df = repo.get_for_parameter(parameter)
    if not df.empty:
        tail = df.sort_values("date", ascending=False).head(12)
        st.dataframe(tail[["date", "batch_id", "formula", "rep1", "rep2", "rep3",
                           "lower_spec", "upper_spec"]],
                     use_container_width=True, hide_index=True)
    else:
        st.info("No data yet. Save a batch above.")
```

```python
import streamlit as st

from data_access.repository import CsvRepository
from spc_engine.control_limits import compute_xbar_r
from spc_engine.capability import compute_capability
from visualization.capability import build_capability_chart


def _get_specs_from_data(df) -> tuple[float, float]:
    """Extract specs from CSV data. Uses the most recent row. NaN = no limit."""
    row = df.iloc[-1]
    return float(row["lower_spec"]), float(row["upper_spec"])


def render_capability_view(repo: CsvRepository, parameter: str):
    st.header(f"📊 Process Capability — {parameter}")

    df = repo.get_for_parameter(parameter)

    if df.empty:
        st.warning(f"No data available for {parameter}. Enter data first.")
        return

    df = df.sort_values("date").reset_index(drop=True)
    limits = compute_xbar_r(df)
    xbar = limits["xbar"]

    import math
    lsl, usl = _get_specs_from_data(df)
    has_lsl = not math.isnan(lsl)
    has_usl = not math.isnan(usl)
    target = (lsl + usl) / 2 if (has_lsl and has_usl) else None

    cap = compute_capability(xbar, lsl, usl)

    # Show which specs are in play
    spec_parts = []
    if has_lsl: spec_parts.append(f"LSL = {lsl:.4g}")
    if has_usl: spec_parts.append(f"USL = {usl:.4g}")
    st.caption(f"Spec limits from data: {', '.join(spec_parts) if spec_parts else 'None'}")

    # KPI row
    col1, col2, col3, col4, col5 = st.columns(5)
    col1.metric("Pp", f"{cap['Pp']:.2f}" if not math.isnan(cap["Pp"]) else "N/A")
    col2.metric("Ppk", f"{cap['Ppk']:.2f}" if not math.isnan(cap["Ppk"]) else "N/A")
    col3.metric("σ_overall", f"{cap['sigma_overall']:.3f}")
    col4.metric("Mean", f"{cap['mean']:.3f}")
    col5.metric("PPM Total", f"{cap['total_ppm']:.0f}")

    # Interpretation
    if math.isnan(cap["Ppk"]):
        st.info("No spec limits defined for this parameter.")
    elif cap["Ppk"] >= 1.33:
        st.success("Process is capable (Ppk ≥ 1.33)")
    elif cap["Ppk"] >= 1.0:
        st.warning("Process is marginally capable (1.0 ≤ Ppk < 1.33)")
    else:
        st.error("Process is not capable (Ppk < 1.0)")

    # Histogram
    fig = build_capability_chart(xbar, lsl, usl, target, parameter)
    st.plotly_chart(fig, use_container_width=True)
```
